# Tracer

## Description

Distributed tracing implementation for monitoring agent execution. Tracks spans, logs, tags, and provides observable streams for real-time monitoring and debugging.

## Signature

```typescript
class Tracer {
  constructor(serviceName?: string)
  startSpan(operationName: string, parentId?: string, tags?: Record<string, string>): TraceContext
  log(spanId: string, message: string, fields?: Record<string, unknown>): void
  setTag(spanId: string, key: string, value: string): void
  endSpan(spanId: string, status?: SpanStatus, error?: Error): void
  getSpan(spanId: string): Span | undefined
  getActiveSpans(): Span[]
  getAllSpans(): Span[]
  observable(): Observable<Span>
  observableByTraceId(traceId: string): Observable<Span>
  getTraceSummary(traceId: string): { spans: Span[], totalDuration: number, failed: number }
  clear(): void
}

function getTracer(serviceName?: string): Tracer
function setTracer(tracer: Tracer): void
```

## Types

### Span

```typescript
interface Span {
  traceId: string;
  spanId: string;
  parentId?: string;
  operationName: string;
  status: SpanStatus;
  startTime: Date;
  endTime?: Date;
  duration?: number;
  tags: Record<string, string>;
  logs: { timestamp: Date; message: string; fields: Record<string, unknown> }[];
}
```

### SpanStatus

```typescript
type SpanStatus = 'started' | 'completed' | 'failed'
```

### TraceContext

```typescript
interface TraceContext {
  traceId: string;
  spanId: string;
  operationName: string;
}
```

## Methods

### startSpan(operationName, parentId, tags)

Start a new span. Creates new traceId if no parent, otherwise inherits from parent.

**Returns:** `TraceContext` - `{ traceId, spanId, operationName }`

### log(spanId, message, fields)

Add log entry to span.

### setTag(spanId, key, value)

Add tag to span for categorization.

### endSpan(spanId, status, error)

End span with status. Calculates duration automatically.

### getTraceSummary(traceId)

Get aggregated trace data.

**Returns:** `{ spans: Span[], totalDuration: number, failed: number }`

## Global Tracer

Module exports singleton `tracer` and `getTracer()` for global access.

```typescript
import { tracer, getTracer } from 'primo-agent';

// Use global instance
tracer.startSpan('my-operation');
```

## Examples

### Example 1: Basic Span Tracking

```typescript
import { getTracer } from 'primo-agent';

const tracer = getTracer('my-service');

const ctx = tracer.startSpan('agent.run');
tracer.log(ctx.spanId, 'Starting agent');
tracer.setTag(ctx.spanId, 'user.id', 'user123');

try {
  // ... agent execution
  tracer.endSpan(ctx.spanId, 'completed');
} catch (e) {
  tracer.endSpan(ctx.spanId, 'failed', e);
}
```

### Example 2: Nested Spans

```typescript
const root = tracer.startSpan('agent.run');
tracer.setTag(root.spanId, 'agent.input', userInput);

// Nested tool span
const tool = tracer.startSpan('tool.calculator', root.spanId);
tracer.log(tool.spanId, 'Executing calculator', { expr });

// ... tool execution

tracer.endSpan(tool.spanId, 'completed');
tracer.endSpan(root.spanId, 'completed');
```

### Example 3: Observable Stream

```typescript
tracer.observableByTraceId(traceId)
  .subscribe((span) => {
    console.log(`${span.operationName}: ${span.status} (${span.duration}ms)`);
  });

// Or filter by status
import { filter } from 'rxjs';

tracer.observable()
  .pipe(filter(s => s.status === 'failed'))
  .subscribe((span) => {
    console.error('Failed span:', span.operationName, span.logs);
  });
```

### Example 4: Trace Summary

```typescript
const summary = tracer.getTraceSummary(traceId);
console.log(`Trace ${traceId}:`);
console.log(`  Spans: ${summary.spans.length}`);
console.log(`  Duration: ${summary.totalDuration}ms`);
console.log(`  Failed: ${summary.failed}`);

for (const span of summary.spans) {
  console.log(`  - ${span.operationName}: ${span.duration}ms`);
}
```
